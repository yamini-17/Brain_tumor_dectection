     "status": "healthy",
     "timestamp": "2024-01-20T10:30:45.123456",
     "model_loaded": true,
     "device": "cuda"
   }

2. GET /status
   Purpose: System configuration and device information
   Returns:
   {
     "system": "Brain Tumor Detection System",
     "version": "1.0.0",
     "model": "YOLOv9",
     "model_path": "model/yolov9.pt",
     "gpu_available": true,
     "gpu_name": "NVIDIA RTX 3090",
     "device": "cuda"
   }

3. POST /predict
   Purpose: Brain tumor detection from MRI image
   Input: multipart/form-data with 'image' field
   Returns:
   {
     "status": "success",
     "tumor_detected": true,
     "confidence": 95.8,
     "bounding_box": [120, 150, 200, 180],
     "processing_time_ms": 23.45,
     "detections_count": 1
